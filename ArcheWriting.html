# find middle divisor of the array data
numbers = []
for i in range(1, len(data) + 1):
  if len(data) % i == 0:
    numbers.append(i)
divisor = numbers[math.ceil(len(numbers) / 2)]

# transform array data into a "rectangle"
grid_values = np.reshape(data, (-1, divisor))

plt.axis('off')
